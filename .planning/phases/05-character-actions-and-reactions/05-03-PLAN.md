      ```typescript
      function toggleCard(index: number) {
        // Don't allow selecting revealed cards
        if (allCards[index].revealed) return
        setSelectedIndices(prev =>
          prev.includes(index)
            ? prev.filter(i => i !== index)
            : prev.length < 2 ? [...prev, index] : prev
        )
      }
      ```

      ```typescript
      function handleConfirm() {
        if (selectedIndices.length !== 2) return
        socket.emit("GAME_ACTION", roomId, {
          type: "EXCHANGE_CHOOSE",
          playerId,
          keptIndices: selectedIndices as [number, number],
        })
      }
      ```

    - Count selectable cards (unrevealed): `const selectableCount = allCards.filter(c => !c.revealed).length`
    - Max to keep: `const maxKeep = Math.min(2, selectableCount)` — normally 2, but if player has only 1 unrevealed card + 2 exchange = 3 selectable, still pick 2.

      ```tsx
        <div className="flex flex-wrap items-center justify-center gap-2">
          {allCards.map((card, index) => {
            if (card.revealed) return null
            const isSelected = selectedIndices.includes(index)
            return (
              <Badge
                key={index}
                variant="secondary"
                className={`cursor-pointer min-h-[44px] px-4 py-2 text-sm ${
                  isSelected ? "ring-2 ring-primary opacity-100" : "opacity-50"

              >
                {card.type}
              </Badge>
            )
      ```
