     ```tsx

       <InfluenceCardSelector myHand={game.myHand} roomId={roomId} playerId={playerId} />
     )}
     ```
   - Show `<CoupTargetSelector>` instead of `<ActionBar>` when `selectingCoupTarget && game.phase === GamePhase.AWAITING_ACTION`:
     ```tsx
     {selectingCoupTarget && game.phase === GamePhase.AWAITING_ACTION ? (
       <CoupTargetSelector
         players={game.players}
         myId={playerId}
         roomId={roomId}
         playerId={playerId}
         onCancel={() => setSelectingCoupTarget(false)}
       />

       <ActionBar ... />
     ```
