EXCHANGE declaration flow (locked decision from ROADMAP + research):
  1. Active player sends { type: "EXCHANGE", playerId }
  2. handleExchange: enters AWAITING_REACTIONS (does NOT touch deck)
     pendingReactions = buildPendingReactions(state, playerId)
  3. Other players PASS or CHALLENGE
  4. When all PASS → resolveAction is called
     resolveAction for "EXCHANGE":
       - draw 2 cards from deck: exchangeCards = [deck.pop()!, deck.pop()!]
       - store in pendingAction.exchangeCards
       - phase = AWAITING_EXCHANGE (NOT AWAITING_ACTION)
       - do NOT advance activePlayerId yet
  5. Active player sends { type: "EXCHANGE_CHOOSE", keptIndices: [i, j] }
     handleExchangeChoose:
       - 4 cards available = [...player.hand, ...pendingAction.exchangeCards!]
       - keptIndices [i, j] select 2 of 4 to keep (indices into the 4-card array)
       - returned cards (the other 2) go back to deck (push onto deck, then shuffleInPlace)
       - player.hand = [kept[0], kept[1]]
       - phase = AWAITING_ACTION, advance turn

  - keptIndices must be [number, number] with both in range [0, 3]
  - keptIndices[0] !== keptIndices[1]
  - Only active player may send EXCHANGE_CHOOSE

```typescript
function handleExchangeChoose(state: GameState, action: Extract<GameAction, { type: "EXCHANGE_CHOOSE" }>): ActionResult {
  if (action.playerId !== state.activePlayerId) {
    return { ok: false, error: "Not your turn" }
  }
  const pending = state.pendingAction!
  const player = state.players.find(p => p.id === action.playerId)!
  const allFour: Card[] = [...player.hand, ...(pending.exchangeCards ?? [])]

  if (allFour.length !== 4) {
    return { ok: false, error: "Exchange cards not available" }
  }

  const [i, j] = action.keptIndices
  if (i === j || i < 0 || i > 3 || j < 0 || j > 3) {
    return { ok: false, error: "Invalid kept indices" }
  }

  const kept: Card[] = [allFour[i], allFour[j]]
  const returned: Card[] = allFour.filter((_, idx) => idx !== i && idx !== j)
  const newDeck = [...state.deck, ...returned]
  shuffleInPlace(newDeck)

  const players = state.players.map(p =>
    p.id === action.playerId ? { ...p, hand: kept } : p
  )

  return {
    ok: true,
    state: {
      ...state,
      players,
      deck: newDeck,
      phase: GamePhase.AWAITING_ACTION,
      activePlayerId: nextActivePlayer(state),
      pendingAction: null,
      log: [...state.log, `${action.playerId} completes Ambassador exchange`],
    },
  }
}
```
