<!-- Nested block-challenge state machine — critical reference for this plan -->

Phase flow (from research deep dive):
  AWAITING_REACTIONS
    → PASS (not last): stay AWAITING_REACTIONS, update pendingReactions[playerId] = "PASSED"
    → PASS (last — all passed): resolve action → AWAITING_ACTION (next turn) or AWAITING_EXCHANGE
    → CHALLENGE: → RESOLVING_CHALLENGE
    → BLOCK: → AWAITING_BLOCK_CHALLENGE (record blockerId, blockerClaimedCard)

  AWAITING_BLOCK_CHALLENGE
    → PASS (not last): stay AWAITING_BLOCK_CHALLENGE, update map
    → PASS (last — all passed): action CANCELLED → AWAITING_ACTION (next turn, no coin change)
    → CHALLENGE: → RESOLVING_BLOCK_CHALLENGE

  RESOLVING_CHALLENGE
    → LOSE_INFLUENCE from acting player (bluffing — challenge succeeded):
        action cancelled → AWAITING_ACTION (next turn advances past acting player)
    → LOSE_INFLUENCE from challenger (failed challenge — challenged player had the card):
        challenger loses card, THEN action resolves (call action resolution inline)
        Note: for TAX/STEAL/ASSASSINATE/EXCHANGE — still stub; for FOREIGN_AID — award 2 coins

  RESOLVING_BLOCK_CHALLENGE
    → LOSE_INFLUENCE from blocker (blocker was bluffing — block-challenge succeeded):
        blocker loses card, original action resolves → AWAITING_ACTION
    → LOSE_INFLUENCE from challenger (block-challenge failed — blocker proved card):
        challenger loses card, action cancelled (block stands) → AWAITING_ACTION

Action resolution helpers needed:
  resolveAction(state, pendingType) → ActionResult
    "FOREIGN_AID": active player coins += 2
    "TAX": active player coins += 3 (stub ok — will be implemented when TAX handler added in Plan 04/05)
    "STEAL": target player coins -= 2, active player coins += 2 (stub ok)
    "EXCHANGE": → phase = AWAITING_EXCHANGE, draw 2 cards from deck into pendingAction.exchangeCards
    others: advance turn, AWAITING_ACTION

pendingReactions tracking for AWAITING_BLOCK_CHALLENGE:
  In AWAITING_BLOCK_CHALLENGE, the reaction window covers players who may challenge the block.
  The active player (original actor) is the primary challenger.
  pendingReactions should be rebuilt to track who must decide on the block challenge.
  Simplest approach: reset pendingReactions to { [activePlayerId]: "WAITING" } when entering AWAITING_BLOCK_CHALLENGE.

LOSE_INFLUENCE in RESOLVING_CHALLENGE — two cases distinguished by who sends it:
  Case A: action.playerId === pendingAction.playerId (acting player loses — challenge succeeded)
    → action cancelled, advance turn to next alive player after acting player
  Case B: action.playerId !== pendingAction.playerId (challenger loses — bluff failed)
    → challenger loses card, then resolve the action as if all had passed
    → call resolveAction(state, pendingAction.type)

function handlePass(state: GameState, action: Extract<GameAction, { type: "PASS" }>): ActionResult {
  const pending = state.pendingAction!
  const updatedReactions = {
    ...pending.pendingReactions,
    [action.playerId]: "PASSED" as const,
  }

  const allDecided = Object.values(updatedReactions).every(r => r !== "WAITING")
  const allPassed = Object.values(updatedReactions).every(r => r === "PASSED")

  if (allDecided && allPassed) {
    const resolvedState = resolveAction({
      ...state,
      pendingAction: { ...pending, pendingReactions: updatedReactions },
    })
    return { ok: true, state: checkGameOver(resolvedState) }
  }

  return {
    ok: true,
    state: {
      ...state,
      pendingAction: { ...pending, pendingReactions: updatedReactions },
    },
  }
}

function handleChallenge(state: GameState, action: Extract<GameAction, { type: "CHALLENGE" }>): ActionResult {
  const pending = state.pendingAction!
  const updatedReactions = {
    ...pending.pendingReactions,
    [action.playerId]: "CHALLENGED" as const,
  }
  return {
    ok: true,
    state: {
      ...state,
      phase: GamePhase.RESOLVING_CHALLENGE,
      pendingAction: { ...pending, pendingReactions: updatedReactions },
      log: [...state.log, `${action.playerId} challenges ${pending.playerId}'s ${pending.type}`],
    },
  }
}

function handleBlock(state: GameState, action: Extract<GameAction, { type: "BLOCK" }>): ActionResult {
  const pending = state.pendingAction!
  // Reset reactions so active player must decide on the block
  const blockChallengeReactions: Record<string, "WAITING"> = {
    [state.activePlayerId]: "WAITING",
  }
  return {
    ok: true,
    state: {
      ...state,
      phase: GamePhase.AWAITING_BLOCK_CHALLENGE,
      pendingAction: {
        ...pending,
        pendingReactions: blockChallengeReactions,
        blockerId: action.playerId,
        blockerClaimedCard: action.claimedCard,
      },
      log: [...state.log, `${action.playerId} blocks with ${action.claimedCard}`],
    },
  }
}
```

function handlePassBlock(state: GameState, action: Extract<GameAction, { type: "PASS" }>): ActionResult {
  const pending = state.pendingAction!
  const updatedReactions = {
    ...pending.pendingReactions,
    [action.playerId]: "PASSED" as const,
  }

  const allPassed = Object.values(updatedReactions).every(r => r === "PASSED")

  if (allPassed) {
    // Block accepted — action is cancelled, advance turn
    return {
      ok: true,
      state: checkGameOver({
        ...state,
        phase: GamePhase.AWAITING_ACTION,
        activePlayerId: nextActivePlayer(state),
        pendingAction: null,
        log: [...state.log, `Block by ${pending.blockerId} accepted — action cancelled`],
      }),
    }
  }

  return {
    ok: true,
    state: {
      ...state,
      pendingAction: { ...pending, pendingReactions: updatedReactions },
    },
  }
}

function handleChallengeBlock(state: GameState, action: Extract<GameAction, { type: "CHALLENGE" }>): ActionResult {
  const pending = state.pendingAction!
  return {
    ok: true,
    state: {
      ...state,
      phase: GamePhase.RESOLVING_BLOCK_CHALLENGE,
      pendingAction: {
        ...pending,
        pendingReactions: { ...pending.pendingReactions, [action.playerId]: "CHALLENGED" as const },
      },
      log: [...state.log, `${action.playerId} challenges ${pending.blockerId}'s block`],
    },
  }
}

function handleLoseInfluenceChallenge(state: GameState, action: Extract<GameAction, { type: "LOSE_INFLUENCE" }>): ActionResult {
  const pending = state.pendingAction!

  const actingPlayerLoses = action.playerId === pending.playerId

  let next = revealCard(state, action.playerId, action.cardIndex)
  next = checkGameOver(next)

  if (next.phase === GamePhase.GAME_OVER) {
    return { ok: true, state: { ...next, pendingAction: null } }
  }

  if (actingPlayerLoses) {
    // Acting player was bluffing — challenge succeeded — action cancelled
    return {
      ok: true,
      state: {
        ...next,
        phase: GamePhase.AWAITING_ACTION,
        activePlayerId: nextActivePlayer(state),
        pendingAction: null,
        log: [...next.log, `${pending.playerId} was bluffing — action cancelled`],
      },
    }
  }

  // Challenger lost — acting player proves their card — action resolves
  const resolvedState = resolveAction({ ...next, pendingAction: pending })
  return { ok: true, state: checkGameOver(resolvedState) }
}

function handleLoseInfluenceBlockChallenge(state: GameState, action: Extract<GameAction, { type: "LOSE_INFLUENCE" }>): ActionResult {
  const pending = state.pendingAction!

  const blockerLoses = action.playerId === pending.blockerId

  let next = revealCard(state, action.playerId, action.cardIndex)
  next = checkGameOver(next)

  if (next.phase === GamePhase.GAME_OVER) {
    return { ok: true, state: { ...next, pendingAction: null } }
  }

  if (blockerLoses) {
    // Blocker was bluffing — block-challenge succeeded — original action resolves
    const resolvedState = resolveAction({ ...next, pendingAction: pending })
    return { ok: true, state: checkGameOver(resolvedState) }
  }

  // Block-challenger lost — block stands — action cancelled
  return {
    ok: true,
    state: {
      ...next,
      phase: GamePhase.AWAITING_ACTION,
      activePlayerId: nextActivePlayer(state),
      pendingAction: null,
      log: [...next.log, `Block by ${pending.blockerId} stands — action cancelled`],
    },
  }
}
```
